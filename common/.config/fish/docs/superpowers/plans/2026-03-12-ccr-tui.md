# ccr TUI Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a telescope-style ratatui fuzzy finder for Claude Code conversation history that searches across all user messages, groups results by conversation, shows preview context, and resumes the selected session.

**Architecture:** Single Rust binary (`ccr`) with two modes: direct (parse JSONL on launch) and daemon (background index over Unix socket). The TUI uses a trait-based data source so it doesn't care which mode provides the data. On selection, prints a shell command to stdout that the fish/zsh wrapper evals.

**Tech Stack:** Rust, ratatui, crossterm, nucleo-matcher, serde_json, notify, tokio (daemon only), clap

**Spec:** `docs/superpowers/specs/2026-03-12-ccr-tui-design.md`

```rust
/// A single user message within a conversation, with its assistant reply.
#[derive(Debug, Clone)]

    pub session_id: String,
    /// Working directory for this conversation
    pub cwd: String,
    /// Project name (basename of cwd)
    pub project_name: String,
    /// File modification time as unix timestamp
    pub timestamp: u64,
    /// All user messages in this conversation

}

/// A reference to a specific message, used as a search result.
#[derive(Debug, Clone)]
pub struct MessageRef {
    /// Index into the conversations vec
    pub conversation_idx: usize,
```

```rust
// src/parser.rs
use crate::types::{Conversation, Message};
use serde_json::Value;
use std::fs;
use std::path::Path;

/// Extract text content from a message's content field.
/// Handles both plain string and array-of-content-blocks formats.
fn extract_text(content: &Value) -> String {
    match content {
        Value::String(s) => s.clone(),
        Value::Array(blocks) => blocks
            .iter()
            .filter_map(|block| {
                if block.get("type")?.as_str()? == "text" {
                    block.get("text")?.as_str().map(|s| s.to_string())
                } else {
                    None
                }
            })
            .collect::<Vec<_>>()
            .join("\n"),
        _ => String::new(),
    }
}

/// Parse a single JSONL file into a Conversation.
/// Returns None if the file has no user messages or can't be read.
pub fn parse_session(path: &Path) -> Option<Conversation> {
    let session_id = path.file_stem()?.to_str()?.to_string();
    let timestamp = path
        .metadata()
        .ok()?
        .modified()
        .ok()?
        .duration_since(std::time::UNIX_EPOCH)
        .ok()?
        .as_secs();

    let content = fs::read_to_string(path).ok()?;
    let mut cwd = String::new();
    let mut messages: Vec<Message> = Vec::new();

    // Collect all records first so we can look ahead for assistant replies.
    let records: Vec<Value> = content
        .lines()
        .filter_map(|line| serde_json::from_str(line).ok())
        .collect();

    let mut msg_index: usize = 0;
    let mut i = 0;
    while i < records.len() {
        let record = &records[i];

        // Extract cwd from the first record that has it.
        if cwd.is_empty() {
            if let Some(c) = record.get("cwd").and_then(|v| v.as_str()) {
                if !c.is_empty() {
                    cwd = c.to_string();
                }
            }
        }

        // Skip meta messages (system-generated, not real user input).
        if record.get("isMeta").and_then(|v| v.as_bool()) == Some(true) {
            i += 1;
            continue;
        }

        // Check for user message.
        if record.get("type").and_then(|v| v.as_str()) == Some("user") {
            if let Some(msg) = record.get("message") {
                if msg.get("role").and_then(|v| v.as_str()) == Some("user") {
                    if let Some(content_val) = msg.get("content") {
                        let text = extract_text(content_val);
                        // Skip empty messages and command messages (e.g. /clear, /compact).
                        if !text.is_empty() && !text.starts_with("<command-name>") {
                            // Look ahead for the next assistant text reply.
                            let reply = find_next_assistant_reply(&records, i + 1);
                            messages.push(Message {
                                index: msg_index,
                                text,
                                reply,
                            });
                            msg_index += 1;
                        }
                    }
                }
            }
        }
        i += 1;
    }

    if messages.is_empty() {
        return None;
    }

    let project_name = Path::new(&cwd)
        .file_name()
        .and_then(|n| n.to_str())
        .unwrap_or("?")
        .to_string();

    Some(Conversation {
        session_id,
        cwd,
        project_name,
        timestamp,
        messages,
    })
}

/// Scan forward from `start` to find the next assistant message with text content.
fn find_next_assistant_reply(records: &[Value], start: usize) -> String {
    for record in &records[start..] {
        if record.get("type").and_then(|v| v.as_str()) == Some("assistant") {
            if let Some(msg) = record.get("message") {
                if msg.get("role").and_then(|v| v.as_str()) == Some("assistant") {
                    if let Some(content_val) = msg.get("content") {
                        let text = extract_text(content_val);
                        if !text.is_empty() {
                            return text;
                        }
                    }
                }
            }
        }
        // Stop if we hit another user message.
        if record.get("type").and_then(|v| v.as_str()) == Some("user") {
            if let Some(msg) = record.get("message") {
                if msg.get("role").and_then(|v| v.as_str()) == Some("user") {
                    break;
                }
            }
        }
    }
    String::new()
}

/// Parse all JSONL files in the Claude projects directory.
/// Returns conversations sorted by mtime descending.
pub fn parse_all_sessions(projects_dir: &Path) -> Vec<Conversation> {
    let pattern = projects_dir.join("*/*.jsonl");
    let pattern_str = pattern.to_str().unwrap_or("");

    let mut conversations: Vec<Conversation> = glob::glob(pattern_str)
        .into_iter()
        .flatten()
        .filter_map(|entry| entry.ok())
        .filter_map(|path| parse_session(&path))
        .collect();

    // Sort by timestamp descending (most recent first).
    conversations.sort_by(|a, b| b.timestamp.cmp(&a.timestamp));
    conversations
}
```

```rust
// src/index.rs
use crate::types::{Conversation, Message, MessageRef};

pub struct SearchIndex {
    conversations: Vec<Conversation>,
}

impl SearchIndex {
    pub fn new(conversations: Vec<Conversation>) -> Self {
        Self { conversations }
    }

    /// Return all message refs (no filtering), for display when query is empty.
    pub fn all_refs(&self) -> Vec<MessageRef> {
        let mut refs = Vec::new();
        for (ci, conv) in self.conversations.iter().enumerate() {
            for (mi, _) in conv.messages.iter().enumerate() {
                refs.push(MessageRef {
                    conversation_idx: ci,
                    message_idx: mi,
                });
            }
        }
        refs
    }

    /// Fuzzy search across all message texts. Returns matching MessageRefs
    /// sorted by match score (best first), with match indices for highlighting.
    pub fn search(&self, query: &str) -> Vec<(MessageRef, Vec<u32>)> {
        if query.is_empty() {
            return self.all_refs().into_iter().map(|r| (r, vec![])).collect();
        }

        use nucleo_matcher::pattern::{CaseMatching, Normalization, Pattern};
        use nucleo_matcher::{Config, Matcher, Utf32Str};

        let pattern = Pattern::parse(query, CaseMatching::Smart, Normalization::Smart);
        let mut buf = Vec::new();
        let mut matcher = Matcher::new(Config::DEFAULT);
        let mut results: Vec<(MessageRef, u32, Vec<u32>)> = Vec::new();

        for (ci, conv) in self.conversations.iter().enumerate() {
            for (mi, msg) in conv.messages.iter().enumerate() {
                let haystack = Utf32Str::new(&msg.text, &mut buf);
                let mut indices = Vec::new();
                if let Some(score) = pattern.indices(haystack, &mut matcher, &mut indices) {
                    results.push((
                        MessageRef {
                            conversation_idx: ci,
                            message_idx: mi,
                        },
                        score,
                        indices,
                    ));
                }
            }
        }

        results.sort_by(|a, b| b.1.cmp(&a.1));
        results.into_iter().map(|(r, _, indices)| (r, indices)).collect()
    }

    pub fn get_conversation(&self, idx: usize) -> &Conversation {
        &self.conversations[idx]
    }

    pub fn get_message(&self, msg_ref: &MessageRef) -> &Message {
        &self.conversations[msg_ref.conversation_idx].messages[msg_ref.message_idx]
    }
}
```

```rust
// src/output.rs

/// Format a shell command to cd into the conversation's directory and resume the session.
/// Uses single-quoting for shell safety.
pub fn format_resume_command(cwd: &str, session_id: &str) -> String {
    format!(
        "cd '{}' && claude --resume '{}'",
        shell_escape(cwd),
        shell_escape(session_id)
    )
}

/// Escape a string for use inside single quotes.
/// The only character that needs escaping in single quotes is the single quote itself.
fn shell_escape(s: &str) -> String {
    s.replace('\'', "'\\''")
}

#[cfg(test)]
```

```rust
#[derive(Parser)]

fn claude_projects_dir() -> PathBuf {
```

```rust
use crate::index::SearchIndex;
use crate::output::format_resume_command;

use crossterm::event::{self, Event, KeyCode, KeyEvent, KeyModifiers};
use crossterm::terminal::{self, EnterAlternateScreen, LeaveAlternateScreen};

use ratatui::backend::CrosstermBackend;

pub struct App {
    index: SearchIndex,
    query: String,
    cursor_pos: usize,
    results: Vec<(MessageRef, Vec<u32>)>,
    selected: usize,
    preview_scroll: u16,
    should_quit: bool,
    selected_result: Option<(String, String)>, // (cwd, session_id)
}

impl App {
    pub fn new(index: SearchIndex) -> Self {
        let results = index.search("");
        Self {
            index,
            query: String::new(),
            cursor_pos: 0,
            results,
            selected: 0,
            preview_scroll: 0,
            should_quit: false,
            selected_result: None,
        }
    }

    fn update_results(&mut self) {
        self.results = self.index.search(&self.query);
        self.selected = 0;
        self.preview_scroll = 0;
    }

    fn handle_key(&mut self, key: KeyEvent) {
        match (key.modifiers, key.code) {
            // Quit
            (_, KeyCode::Esc) => self.should_quit = true,
            (KeyModifiers::CONTROL, KeyCode::Char('c')) => self.should_quit = true,

            // Navigate
            (KeyModifiers::CONTROL, KeyCode::Char('n')) | (_, KeyCode::Down) => {
                if !self.results.is_empty() {
                    self.selected = (self.selected + 1).min(self.results.len() - 1);
                    self.preview_scroll = 0;
                }
            }
            (KeyModifiers::CONTROL, KeyCode::Char('p')) | (_, KeyCode::Up) => {
                self.selected = self.selected.saturating_sub(1);
                self.preview_scroll = 0;
            }

            // Scroll preview
            (KeyModifiers::CONTROL, KeyCode::Char('d')) => {
                self.preview_scroll = self.preview_scroll.saturating_add(5);
            }
            (KeyModifiers::CONTROL, KeyCode::Char('u')) => {
                self.preview_scroll = self.preview_scroll.saturating_sub(5);
            }

            // Accept
            (_, KeyCode::Enter) | (KeyModifiers::CONTROL, KeyCode::Char('y')) => {
                if let Some((ref msg_ref, _)) = self.results.get(self.selected) {
                    let conv = self.index.get_conversation(msg_ref.conversation_idx);
                    self.selected_result =
                        Some((conv.cwd.clone(), conv.session_id.clone()));
                }
                self.should_quit = true;
            }

            // Typing
            (_, KeyCode::Char(c)) if key.modifiers.is_empty() || key.modifiers == KeyModifiers::SHIFT => {
                // Find the byte offset for the character position.
                let byte_pos = self.query.char_indices()
                    .nth(self.cursor_pos)
                    .map(|(i, _)| i)
                    .unwrap_or(self.query.len());
                self.query.insert(byte_pos, c);
                self.cursor_pos += 1;
                self.update_results();
            }
            (_, KeyCode::Backspace) => {
                if self.cursor_pos > 0 {
                    self.cursor_pos -= 1;
                    let byte_pos = self.query.char_indices()
                        .nth(self.cursor_pos)
                        .map(|(i, _)| i)
                        .unwrap_or(self.query.len());
                    self.query.remove(byte_pos);
                    self.update_results();
                }
            }

            _ => {}
        }
    }
}

pub fn run(index: SearchIndex) -> io::Result<Option<String>> {
    // Setup terminal
    terminal::enable_raw_mode()?;
    let mut stderr = io::stderr();
    execute!(stderr, EnterAlternateScreen)?;
    let backend = CrosstermBackend::new(stderr);
    let mut terminal = Terminal::new(backend)?;

    let mut app = App::new(index);

    // Main loop
    loop {
        terminal.draw(|f| {
            let chunks = Layout::default()
                .direction(Direction::Vertical)
                .constraints([
                    Constraint::Length(3),  // search input
                    Constraint::Min(1),     // results + preview
                    Constraint::Length(1),  // status bar
                ])
                .split(f.area());

            // Search input
            input::render(f, chunks[0], &app.query, app.cursor_pos);

            // Results + Preview (horizontal split)
            let main_chunks = Layout::default()
                .direction(Direction::Horizontal)
                .constraints([Constraint::Percentage(45), Constraint::Percentage(55)])
                .split(chunks[1]);

            results::render(f, main_chunks[0], &app.index, &app.results, app.selected);

            // Preview pane
            if let Some((ref msg_ref, ref indices)) = app.results.get(app.selected) {
                preview::render(
                    f,
                    main_chunks[1],
                    &app.index,
                    msg_ref,
                    indices,
                    app.preview_scroll,
                );
            }

            // Status bar
            let status = if app.results.is_empty() {
                " no matches".to_string()
            } else if let Some((ref msg_ref, _)) = app.results.get(app.selected) {
                let conv = app.index.get_conversation(msg_ref.conversation_idx);
                format!(
                    " {}/{} matches │ {} │ {} │ ⏎ resume",
                    app.selected + 1,
                    app.results.len(),
                    conv.project_name,
                    conv.cwd
                )
            } else {
                String::new()
            };
            let status_widget = ratatui::widgets::Paragraph::new(status)
                .style(
                    ratatui::style::Style::default()
                        .bg(ratatui::style::Color::DarkGray)
                        .fg(ratatui::style::Color::White),
                );
            f.render_widget(status_widget, chunks[2]);
        })?;

        if app.should_quit {
            break;
        }

        if event::poll(std::time::Duration::from_millis(50))? {
            if let Event::Key(key) = event::read()? {
                app.handle_key(key);
            }
        }
    }

    // Restore terminal
    terminal::disable_raw_mode()?;
    execute!(terminal.backend_mut(), LeaveAlternateScreen)?;

    // Return the resume command if a selection was made
    Ok(app
        .selected_result
        .map(|(cwd, sid)| format_resume_command(&cwd, &sid)))
}
```

```rust
use crate::index::SearchIndex;
use crate::types::MessageRef;
use ratatui::layout::Rect;
use ratatui::style::{Color, Modifier, Style};
use ratatui::text::{Line, Span};
use ratatui::widgets::{Block, Borders, List, ListItem};
use ratatui::Frame;

pub fn render(
    f: &mut Frame,
    area: Rect,
    index: &SearchIndex,
    results: &[(MessageRef, Vec<u32>)],
    selected: usize,
) {
    let mut items: Vec<ListItem> = Vec::new();
    let mut current_conv_idx: Option<usize> = None;
    let mut visual_selected: usize = 0;

    for (ri, (msg_ref, match_indices)) in results.iter().enumerate() {
        // Insert conversation header when conversation changes.
        if current_conv_idx != Some(msg_ref.conversation_idx) {
            current_conv_idx = Some(msg_ref.conversation_idx);
            let conv = index.get_conversation(msg_ref.conversation_idx);
            let date = chrono_lite_format(conv.timestamp);
            let header = format!("── {} ({}) ──", conv.project_name, date);
            items.push(ListItem::new(Line::from(Span::styled(
                header,
                Style::default()
                    .fg(Color::DarkGray)
                    .add_modifier(Modifier::BOLD),
            ))));
            // header is not selectable
        }

        if ri == selected {
            visual_selected = items.len();
        }

        let msg = index.get_message(msg_ref);
        let truncated: String = msg.text.chars().take(80).collect();
        let prefix = format!(" {:>2}: ", msg.index);

        // Build spans with match highlighting.
        let mut spans = vec![Span::styled(
            prefix,
            Style::default().fg(Color::Magenta),
        )];

        let match_set: std::collections::HashSet<u32> =
            match_indices.iter().copied().collect();

        for (ci, ch) in truncated.chars().enumerate() {
            let style = if match_set.contains(&(ci as u32)) {
                Style::default()
                    .fg(Color::Yellow)
                    .add_modifier(Modifier::BOLD)
            } else {
                Style::default().fg(Color::White)
            };
            spans.push(Span::styled(ch.to_string(), style));
        }

        let is_selected = ri == selected;
        let mut item = ListItem::new(Line::from(spans));
        if is_selected {
            item = item.style(
                Style::default()
                    .bg(Color::Rgb(40, 52, 87))
                    .fg(Color::White),
            );
        }
        items.push(item);
    }

    // Calculate offset to keep selected item visible.
    let visible_height = area.height.saturating_sub(2) as usize; // minus borders
    let offset = if visual_selected >= visible_height {
        visual_selected - visible_height + 1
    } else {
        0
    };

    let list = List::new(items.into_iter().skip(offset).collect::<Vec<_>>())
        .block(
            Block::default()
                .borders(Borders::ALL)
                .border_style(Style::default().fg(Color::DarkGray))
                .title(" Results "),
        );

    f.render_widget(list, area);
}

/// Simple timestamp to "MM/DD HH:MM" without pulling in chrono.
fn chrono_lite_format(timestamp: u64) -> String {
    let secs = timestamp as i64;
    let days = secs / 86400;
    let time_of_day = secs % 86400;
    let hours = time_of_day / 3600;
    let minutes = (time_of_day % 3600) / 60;

    // Simplified date calculation (good enough for display purposes).
    // This gives us month/day from a unix timestamp.
    let (year, month, day) = unix_days_to_date(days as i32);
    format!("{:02}/{:02} {:02}:{:02}", month, day, hours, minutes)
}

fn unix_days_to_date(days: i32) -> (i32, u32, u32) {
    // Algorithm from http://howardhinnant.github.io/date_algorithms.html
    let z = days + 719468;
    let era = if z >= 0 { z } else { z - 146096 } / 146097;
    let doe = (z - era * 146097) as u32;
    let yoe = (doe - doe / 1460 + doe / 36524 - doe / 146096) / 365;
    let y = yoe as i32 + era * 400;
    let doy = doe - (365 * yoe + yoe / 4 - yoe / 100);
    let mp = (5 * doy + 2) / 153;
    let d = doy - (153 * mp + 2) / 5 + 1;
    let m = if mp < 10 { mp + 3 } else { mp - 9 };
    let y = if m <= 2 { y + 1 } else { y };
    (y, m, d)
}
```

```rust
#[derive(Serialize, Deserialize)]
struct ConversationWire {

    cwd: String,
    project_name: String,
    timestamp: u64,
    messages: Vec<MessageWire>,

#[derive(Serialize, Deserialize)]
struct MessageWire {
    index: usize,
    text: String,
    reply: String,
}

impl From<&Conversation> for ConversationWire {
    fn from(conv: &Conversation) -> Self {
        ConversationWire {
            session_id: conv.session_id.clone(),
            cwd: conv.cwd.clone(),
            project_name: conv.project_name.clone(),
            timestamp: conv.timestamp,
            messages: conv
                .messages
                .iter()
                .map(|m| MessageWire {
                    index: m.index,
                    text: m.text.clone(),
                    reply: m.reply.clone(),
                })
                .collect(),
        }
    }
}

pub fn socket_path() -> PathBuf {
    if let Ok(runtime_dir) = std::env::var("XDG_RUNTIME_DIR") {
        PathBuf::from(runtime_dir).join("ccr.sock")
    } else {
        let uid = unsafe { libc::getuid() };
        PathBuf::from(format!("/tmp/ccr-{}.sock", uid))
    }
}

pub fn pid_path() -> PathBuf {
    let mut p = socket_path();
    p.set_extension("pid");
    p
}

#[tokio::main]
pub async fn start(projects_dir: PathBuf) {
    let sock = socket_path();
    // Clean up stale socket.
    let _ = std::fs::remove_file(&sock);

    // Write PID file.
    std::fs::write(pid_path(), std::process::id().to_string()).expect("failed to write PID file");

    // Initial parse.
    let conversations = Arc::new(RwLock::new(parser::parse_all_sessions(&projects_dir)));
    eprintln!(
        "ccr daemon: loaded {} conversations",
        conversations.read().unwrap().len()
    );

    // File watcher with debouncing (2 second delay to batch rapid writes).
    let convs_for_watcher = Arc::clone(&conversations);
    let projects_dir_for_watcher = projects_dir.clone();
    let mut debouncer = new_debouncer(
        Duration::from_secs(2),
        move |res: Result<Vec<notify_debouncer_mini::DebouncedEvent>, notify::Error>| {
            if res.is_ok() {
                let new_convs = parser::parse_all_sessions(&projects_dir_for_watcher);
                *convs_for_watcher.write().unwrap() = new_convs;
                eprintln!("ccr daemon: reloaded index");
            }
        },
    )
    .expect("failed to create file watcher");

    debouncer
        .watcher()
        .watch(&projects_dir, notify::RecursiveMode::Recursive)
        .expect("failed to watch projects directory");

    // Unix socket server.
    let listener = UnixListener::bind(&sock).expect("failed to bind socket");
    eprintln!("ccr daemon: listening on {}", sock.display());

    loop {
        if let Ok((stream, _)) = listener.accept().await {
            let convs = Arc::clone(&conversations);
            tokio::spawn(async move {
                let (reader, mut writer) = stream.into_split();
                let mut buf_reader = BufReader::new(reader);
                let mut line = String::new();

                if buf_reader.read_line(&mut line).await.is_ok() {
                    let line = line.trim();
                    if line.starts_with("DUMP") {
                        // Collect all JSON while holding the lock, then drop it before async writes.
                        let json_lines: Vec<String> = {
                            let convs = convs.read().unwrap();
                            convs.iter()
                                .filter_map(|conv| {
                                    let wire: ConversationWire = conv.into();
                                    serde_json::to_string(&wire).ok()
                                })
                                .collect()
                        }; // RwLock guard dropped here

                        for json in &json_lines {
                            let _ = writer.write_all(format!("{}\n", json).as_bytes()).await;
                        }
                        let _ = writer.write_all(b"END\n").await;
                    }
                }
            });
        }
    }
}

pub fn stop() {
    let pid_file = pid_path();
    if let Ok(pid_str) = std::fs::read_to_string(&pid_file) {
        if let Ok(pid) = pid_str.trim().parse::<i32>() {
            unsafe {
                libc::kill(pid, libc::SIGTERM);
            }
            let _ = std::fs::remove_file(pid_file);
            let _ = std::fs::remove_file(socket_path());
            eprintln!("ccr daemon: stopped (pid {})", pid);
        }
    } else {
        eprintln!("ccr daemon: not running");
    }
}
```

```rust
// src/daemon/client.rs
use crate::types::{Conversation, Message};
use std::io::{BufRead, BufReader, Write};
use std::os::unix::net::UnixStream;
use std::path::Path;
use std::time::Duration;

use serde::Deserialize;

#[derive(Deserialize)]
struct ConversationWire {
    session_id: String,
    cwd: String,
    project_name: String,
    timestamp: u64,
    messages: Vec<MessageWire>,
}

#[derive(Deserialize)]
struct MessageWire {
    index: usize,
    text: String,
    reply: String,
}

/// Try to connect to the daemon and fetch all conversations.
/// Returns None if the daemon is not running or connection fails.
pub fn fetch_from_daemon(socket_path: &Path) -> Option<Vec<Conversation>> {
    let mut stream = UnixStream::connect(socket_path).ok()?;
    stream
        .set_read_timeout(Some(Duration::from_secs(5)))
        .ok()?;

    stream.write_all(b"DUMP v1\n").ok()?;

    let reader = BufReader::new(stream);
    let mut conversations = Vec::new();

    for line in reader.lines() {
        let line = line.ok()?;
        if line == "END" {
            break;
        }
        if let Ok(wire) = serde_json::from_str::<ConversationWire>(&line) {
            conversations.push(Conversation {
                session_id: wire.session_id,
                cwd: wire.cwd,
                project_name: wire.project_name,
                timestamp: wire.timestamp,
                messages: wire
                    .messages
                    .into_iter()
                    .map(|m| Message {
                        index: m.index,
                        text: m.text,
                        reply: m.reply,
                    })
                    .collect(),
            });
        }
    }

    Some(conversations)
}
```
